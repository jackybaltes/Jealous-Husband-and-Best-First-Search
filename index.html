Jealous Husband Problem

  • There are three couples that want to cross a river
  • There is a small boat which can take only two people at a time
  • The husbands are very jealous and do not want to leave their wives in the presence of another husband without them being there
  • Help them cross the river
BOAT,H1,W1,H2,W2,H3,W3, NUM_OBJECTS = 0,1,2,3,4,5,6,7
LEFT, RIGHT = 0, 1

initial = [ LEFT for i in range(NUM_OBJECTS) ]

objNames = [ "Boat", "Husband 1", "Wife 1", "Husband 2", "Wife 2", "Husband 3", "Wife 3" ]

def isValid( state ):
    result = True
    for w in [ W1, W2, W3 ]:
        if state[w] != state[w-1]:
            for h in [ H1, H2, H3 ]:
                if h != w - 1:
                    if state[h] == state[w]:
                        result = False
    return result

def jealousHusbandsSuccessors( state, hist = None ):
    children = []
    for p1 in range(1,NUM_OBJECTS):
        if state[p1] == state[BOAT]:
            ns = state.copy()
            ns[p1] = 1 - state[p1]
            ns[BOAT] = 1 - state[BOAT]
            if isValid(ns):
                children.append( ns )
            for p2 in range( p1+1, NUM_OBJECTS ):
                if state[p1] == state[p2]:
                    ns = state.copy()
                    ns[p1] = 1 - state[p1]
                    ns[p2] = 1 - state[p2]
                    ns[BOAT] = 1 - state[BOAT]
                    if isValid( ns ):
                        children.append(ns)
    return children

def stateToStr( state ):
    leftObjs = []
    rightObjs = []

    for o in range(len(state)):
        if state[o] == LEFT:
            leftObjs.append( objNames[o] )
        else:
            rightObjs.append( objNames[o] )
    return "[" + ",".join( leftObjs) + "],[" + ",".join( rightObjs) + "]"

def jealousHusbandsIsGoal( state ):
    result = True
    for o in range(len(state)):
        if state[o] != RIGHT:
            result = False
            break
    return result

def search( initial, successorFunc, isGoalFunc, dfs = True, limit = 1000 ):
    queue = [ ( initial, [initial] ) ]
    numNodes = 1
    maxQueue = 0
    sol = None
    print("Search started", queue)
    while( len( queue ) > 0 ):
        # current, chist = queue[-1]
        # queue = queue[0:-1]
        #print("Q", queue)
        if ( len(queue) > maxQueue ):
              maxQueue = len(queue)

        current, chist = queue.pop()
        
        #print("Num Nodes", numNodes)
        #stateToStr( current )
        if isGoalFunc( current ):
            sol = (numNodes, maxQueue, chist)
            break
        #print("current state", current, chist )
        #print("queue", queue)
        children = successorFunc( current )
        #print("children", children)
        for c in children:
            if c not in chist:
                #print("Add child", stateToStr(c))
                if dfs:
                  insert_index = len(queue)
                else:
                  insert_index = 0

                queue.insert( insert_index, (c, chist + [c] ))
                numNodes = numNodes + 1
                if ( numNodes > limit ):
                    print("Search terminated with numNodes", numNodes )
                    return None
    return sol

sol = search( initial, jealousHusbandsSuccessors, jealousHusbandsIsGoal, True, 10000 )
if sol:

    print('Solution', nNodes, 'length', len(trace), 'max queue size', maxQueue )
    print("Solution Trace" )
    for s in trace:
        print( stateToStr( s ) )
    print("End of Solution")

Heuristic Function for Jealous Husband Problem

  • Think about minimum number of moves necessary to solve the puzzle
  • Since the boat can only take two people
    • I need one move to move two people with the boat
    • I need three moves to move three people with the boat
    • I need 2*n - 1 moves for n people

Heuristic Function for Jealous Husband Problem

  • We can also consider the position of the boat
  • Three moves need to be added if the boat is on the wrong side of the river
